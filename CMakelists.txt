cmake_minimum_required(VERSION 3.20)
project(E1Morph VERSION 1.0.0)

# ==========================================
# 1. 依存ライブラリの自動取得 (JUCE & Eigen)
# ==========================================
include(FetchContent)

# JUCE 8.0.x を取得
FetchContent_Declare(juce
    GIT_REPOSITORY https://github.com/juce-framework/JUCE.git
    GIT_TAG        8.0.4
)
FetchContent_MakeAvailable(juce)

# Eigen 3.4.0 を取得（ヘッダーオンリーライブラリ）
FetchContent_Declare(eigen
    GIT_REPOSITORY https://gitlab.com/libeigen/eigen.git
    GIT_TAG        3.4.0
)
FetchContent_MakeAvailable(eigen)

# ==========================================
# 2. プラグインの定義
# ==========================================
juce_add_plugin(E1Morph
    COMPANY_NAME "MyCompany"          # 任意の会社名
    PLUGIN_MANUFACTURER_CODE "MyCo"   # 4文字のメーカーコード
    PLUGIN_CODE "E1Mo"                # 4文字のプラグインコード
    FORMATS VST3 Standalone           # ★Standaloneを入れるとデバッグが超楽です
    PRODUCT_NAME "E1Morph"
)

# ==========================================
# 3. ソースファイルと機能のリンク
# ==========================================
target_sources(E1Morph PRIVATE
    Source/PluginProcessor.cpp
    Source/PluginProcessor.h
)

target_compile_features(E1Morph PRIVATE cxx_std_20)

# JUCEモジュールとEigenをリンク
target_link_libraries(E1Morph PRIVATE
    juce::juce_audio_utils
    juce::juce_dsp
    Eigen3::Eigen
)
